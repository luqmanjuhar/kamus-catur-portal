    <script>
        const googleAppScriptURL = 'https://script.google.com/macros/s/AKfycbws8S8vbGeamIPWcRiAcH0o1awq6aSZkCLOWb1_Tng4/dev'; 

        window.onload = () => {
            const userId = sessionStorage.getItem('userId');
            const userRole = sessionStorage.getItem('userRole');

            if (!userId || userRole !== 'student') {
                document.getElementById('loading-spinner').innerHTML = '<p class="text-red-500">Access Denied. Please <a href="login.html" class="font-bold text-primary">log in</a> as a student.</p>';
                // In a live environment, this would redirect:
                // window.location.href = 'login.html'; 
            } else {
                fetchStudentData(userId);
                setupTabEvents(userId);
            }

            document.getElementById('logout-button').addEventListener('click', () => {
                sessionStorage.clear();
                window.location.href = 'login.html';
            });
        };

        async function fetchStudentData(studentId) {
            try {
                const response = await fetch(googleAppScriptURL, {
                    method: 'POST',
                    body: JSON.stringify({ action: 'getProfile', userId: studentId, role: 'student' }),
                    headers: { 'Content-Type': 'application/json' },
                });

                const result = await response.json();

                if (result.success && result.data) {
                    populateProfile(result.data);
                    document.getElementById('loading-spinner').classList.add('hidden');
                    document.getElementById('dashboard-content').classList.remove('hidden');
                } else {
                    throw new Error(result.message || 'Could not retrieve profile.');
                }
            } catch (error) {
                console.error('Failed to fetch student data:', error);
                document.getElementById('loading-spinner').innerHTML = `<p class="text-red-500">Error loading profile: ${error.message}. Please try logging in again.</p>`;
            }
        }

        function populateProfile(data) {
            document.getElementById('student-name-header').textContent = data.name || 'Student';
            
            if (data.profilePhotos && data.profilePhotos.startsWith('http')) {
                document.getElementById('profile-photo').src = data.profilePhotos;
            }
            document.getElementById('student-name').textContent = data.name || '-';
            document.getElementById('student-id').textContent = `ID: ${data.studentId || '-'}`;
            const statusEl = document.getElementById('status');
            statusEl.textContent = data.status || 'Unknown';
            if (data.status && data.status.toLowerCase() === 'active') {
                statusEl.className = 'mt-2 text-xs font-semibold px-3 py-1 rounded-full bg-green-100 text-green-800';
            } else {
                 statusEl.className = 'mt-2 text-xs font-semibold px-3 py-1 rounded-full bg-red-100 text-red-800';
            }
            document.getElementById('lichess-username').textContent = data.lichessUsername || '-';
            document.getElementById('identity-no').textContent = data.identityNo || '-';
            document.getElementById('email').textContent = data.email || '-';
            document.getElementById('contact-no').textContent = data.contactNo || '-';
            document.getElementById('school-name').textContent = data.schoolName || '-';
            document.getElementById('day-of-class').textContent = data.dayOfClass || '-';
            document.getElementById('type-of-class').textContent = data.typeOfClass || '-';
            document.getElementById('level').textContent = data.level || '-';
            document.getElementById('assigned-coach').textContent = data.assignedCoach || '-';
        }
        
        function setupTabEvents(studentId) {
            const tabs = document.querySelectorAll('.tab');
            const tabContent = document.getElementById('tab-content');
            
            // Load initial tab content
            tabContent.innerHTML = `<p class="text-gray-600">Select a tab to view details.</p>`;
            document.querySelector('.tab[data-tab="attendance"]').click();

            tabs.forEach(tab => {
                tab.addEventListener('click', () => {
                    tabs.forEach(t => t.classList.remove('tab-active'));
                    tab.classList.add('tab-active');
                    const tabName = tab.dataset.tab;
                    tabContent.innerHTML = `<div class="text-center py-5"><div class="animate-spin rounded-full h-8 w-8 border-b-2 border-primary mx-auto"></div></div>`;
                    
                    if (tabName === 'attendance') {
                        // We will add the function to fetch attendance data later
                        tabContent.innerHTML = `<p class="text-gray-600">Attendance records will be displayed here.</p>`;
                    } else if (tabName === 'performance') {
                        tabContent.innerHTML = `<p class="text-gray-600">Performance records will be displayed here.</p>`;
                    } else if (tabName === 'tournaments') {
                        tabContent.innerHTML = `<p class="text-gray-600">Tournament records will be displayed here.</p>`;
                    }
                });
            });
        }
